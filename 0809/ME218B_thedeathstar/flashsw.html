void initFlash(void)
    Set the data direction of the flash sensor pin to be an input

char getFlash(void)
    If the pin is high
        Return “flash is on”
    Else
        Return “flash is off”

void main(void) **Test Harness**
    Initialize the flash module
    Repeat forever
        Check for a flash
        Report a positive result
